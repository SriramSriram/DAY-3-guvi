Qn 1:
For the given JSON iterate over all for loops (for, for in, for of, forEach)

CODE:
var obj= [
    {person:"sri", age:20, company:"Guvi"},
    {person:"ram", age:25, company:"Guvi Geek"},
    {person: "sekar", age:35,company:"Guvi Geek Academy"}
]

// using for loop:
for(i=0; i<obj.length; i++){
    user=obj[i]
    console.log(`
        Name: ${user.person}
        Age: ${user.age}
        Company: ${user.company}
        `
    );
}

// using forEach loop:
obj.forEach((user)=> {
    console.log(`
        Name: ${user.person}
        Age: ${user.age}
        Company: ${user.company}
        `
    );

})

// using forIn loop:

for (var index in obj) {
    user=obj[index];
    console.log(`
        Name: ${user.person}
        Age: ${user.age}
        Company: ${user.company}
        `
    );

}

// using forOf loop:

for(var index of obj){
    console.log(`
        Name: ${index.person}
        Age; ${index.age}
        Company: ${index.company}
        `
        );
}
